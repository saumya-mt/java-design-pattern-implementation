package sequentialSolution;

/**
 * Entry point for running the sequential processing of OULAD data. This program processes student
 * activity data from CSV files sequentially.
 */
public class SequentialMain {

  /**
   * Main method to run the program.
   *
   * @param args Command-line arguments: - input_directory: The directory containing courses.csv and
   *             studentVle.csv.
   */

  public static void main(String[] args) {
    if (args.length != 1) {
      System.out.println("Usage: java SequentialProcessor <input_directory>");
      System.out.println("  input_directory: Directory containing courses.csv and studentVle.csv");
      System.exit(1);
    }

    String inputDirectory = args[0];
    new SequentialProcessor(inputDirectory).execute();
  }

}
