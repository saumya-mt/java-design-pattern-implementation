package concurrentSolution;

/**
 * Entry point for the concurrent processing of OULAD data.
 */
public class ConcurrentMain {

  /**
   * Main method to run the program.
   *
   * @param args Command-line arguments (expects one argument: the input directory path).
   */
  public static void main(String[] args) {
    if (args.length != 1) {
      System.out.println("Usage: java ConcurrentProcessor <input_directory>");
      System.out.println("  input_directory: Directory containing courses.csv and studentVle.csv");
      System.exit(1);
    }

    String inputDirectory = args[0];
    new ConcurrentProcessor(inputDirectory).execute();
  }

}
