package concurrentSolution;

import java.io.IOException;

/**
 * Entry point for processing activity data based on a threshold. This program filters and processes
 * summary CSV files generated by either ConcurrentProcessor or SequentialProcessor.
 */
public class ThresholdMain {

  /**
   * Main method to run the threshold processing.
   *
   * @param args Command line arguments: output_directory threshold
   */
  public static void main(String[] args) {
    if (args.length != 2) {
      System.out.println("Usage: java ThresholdMain <output_directory> <threshold>");
      System.out.println("  output_directory: Directory containing the summary CSV files");
      System.out.println("  threshold: Minimum number of clicks to identify high activity days");
      System.out.println(
          "\nNote: Make sure to run either ConcurrentProcessor or SequentialProcessor first");
      System.out.println("      to generate the summary files in the output directory.");
      System.exit(1);
    }

    String outputDirectory = args[0];
    int threshold;
    try {
      threshold = Integer.parseInt(args[1]);
      if (threshold < 0) {
        throw new IllegalArgumentException("Threshold must be non-negative");
      }
    } catch (NumberFormatException e) {
      System.out.println("Invalid threshold value. Please provide a valid integer.");
      return;
    }

    try {
      ThresholdProcessor processor = new ThresholdProcessor(outputDirectory, threshold);
      processor.processFiles();
    } catch (IOException e) {
      System.err.println("Error: " + e.getMessage());
      System.exit(1);
    }
  }

}
