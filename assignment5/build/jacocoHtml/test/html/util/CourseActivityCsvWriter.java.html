package util;

import java.io.*;
import java.util.*;

/**
 * Utility class for writing CSV files containing processed OULAD data. This class handles writing
 * individual course activity files.
 */
public class CourseActivityCsvWriter {

  /**
   * Writes individual CSV files for each course's activity data. Each file contains the total
   * clicks per day for a specific course presentation.
   *
   * @param outputDirectory output directory
   * @param data            Map of course keys to maps of dates and total clicks
   * @throws IOException if there is an error writing the files
   */
  public static void writeSummaryFiles(String outputDirectory,
      Map<String, Map<Integer, Integer>> data) throws IOException {
    if (data == null || data.isEmpty()) {
      return;
    }

    for (Map.Entry<String, Map<Integer, Integer>> entry : data.entrySet()) {
      String courseKey = entry.getKey();
      Map<Integer, Integer> activityData = entry.getValue();

      if (activityData == null || activityData.isEmpty()) {
        continue;
      }

      String fileName = outputDirectory + "/" + courseKey + ".csv";
      try (BufferedWriter writer = new BufferedWriter(new FileWriter(fileName))) {
        // Write header
        writer.write("date,total_clicks");
        writer.newLine();

        // Write data
        for (Map.Entry<Integer, Integer> activity : activityData.entrySet()) {
          writer.write(activity.getKey() + "," + activity.getValue());
          writer.newLine();
        }
      }
    }
  }
}
