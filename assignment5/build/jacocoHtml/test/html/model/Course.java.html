package model;

import java.util.Objects;

/**
 * Represents a course in the OULAD dataset.
 * This class is immutable and represents a course module with its presentation details.
 */
public class Course {
  private final String codeModule;
  private final String codePresentation;
  private final int presentationLength;

  /**
   * Constructs a new Course instance.
   *
   * @param codeModule The module code (e.g., "CS5010")
   * @param codePresentation The presentation code (e.g., "Spring2023")
   * @param presentationLength The length of the course in days
   */
  public Course(String codeModule, String codePresentation, int presentationLength) {
    this.codeModule = codeModule;
    this.codePresentation = codePresentation;
    this.presentationLength = presentationLength;
  }

  /**
   * Gets the module code of the course.
   *
   * @return The module code
   */
  public String getCodeModule() {
    return codeModule;
  }

  /**
   * Gets the presentation code of the course.
   *
   * @return The presentation code
   */
  public String getCodePresentation() {
    return codePresentation;
  }

  /**
   * Gets the length of the course presentation in days.
   *
   * @return The presentation length in days
   */
  public int getPresentationLength() {
    return presentationLength;
  }

  @Override
  public boolean equals(Object o) {
    if (this == o) return true;
    if (o == null || getClass() != o.getClass()) return false;
    Course course = (Course) o;
    return presentationLength == course.presentationLength &&
        Objects.equals(codeModule, course.codeModule) &&
        Objects.equals(codePresentation, course.codePresentation);
  }

  @Override
  public int hashCode() {
    return Objects.hash(codeModule, codePresentation, presentationLength);
  }

  @Override
  public String toString() {
    return "Course{" +
        "codeModule='" + codeModule + '\'' +
        ", codePresentation='" + codePresentation + '\'' +
        ", presentationLength=" + presentationLength +
        '}';
  }
}
